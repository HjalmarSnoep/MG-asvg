This is a demonstration of how to use javascript in conjunction with ASVG. These buttons will animate on pointerover.

Due to this script: 

          
document.getElementById("button1").addEventListener("mouseover",mouseOver);
document.getElementById("button1").addEventListener("mouseout",mouseOut);
document.getElementById("button2").addEventListener("mouseover",mouseOver);
document.getElementById("button2").addEventListener("mouseout",mouseOut);
function mouseOver(ev)
{
 if(ev.currentTarget.id=="button1")
 {
    ASVG.startTimeline("button1-timeline");
 }
 if(ev.currentTarget.id=="button2")
 {
    ASVG.startTimeline("button2-timeline");
 }
}
function mouseOut(ev)
{
 ASVG.stopTimeline("button1-timeline");
 ASVG.stopTimeline("button2-timeline");
}
ASVG.stopTimeline("button1-timeline");
ASVG.stopTimeline("button2-timeline");
